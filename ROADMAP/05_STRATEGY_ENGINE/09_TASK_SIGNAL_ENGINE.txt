@dataclass
class TradeSetup:
    symbol: str
    direction: str                     # "BUY" or "SELL"
    
    # Entry information
    entry_zone_high: float
    entry_zone_low: float
    entry_target: float                # Mid of OB or FVG
    
    # SL/TP (calculated later by Risk Engine)
    suggested_sl: float                # Based on structure
    suggested_tp: float                # Based on liquidity target
    
    # Timeframe context
    h4_bias: str                       # "BULLISH", "BEARISH", "NEUTRAL"
    h4_trend: str
    h4_regime: MarketRegime
    h1_structure_aligned: bool
    h1_bos_direction: str              # "BULLISH" or "BEARISH"
    m15_setup_type: str                # "OB", "FVG", "OB+FVG"
    m15_liquidity_swept: bool
    m15_order_block: Optional[OrderBlock]
    m15_fvg: Optional[FairValueGap]
    m5_confirmation: bool
    m5_confirmation_type: str          # "BOS", "CHoCH", "DISPLACEMENT"
    
    # Confluence factors (used by scoring engine)
    has_h4_bias: bool
    has_h1_structure: bool
    has_bos_choch: bool
    has_liquidity_sweep: bool
    has_valid_ob: bool
    has_valid_fvg: bool
    has_m5_confirmation: bool
    has_ema_alignment: bool

    has_valid_rr: bool                 # Set by risk engine
    
    # Metadata
    signal_id: str                     # UUID for deduplication

    spread: float

    def analyze_symbol(self, symbol: str) -> Optional[TradeSetup]:
        """
        Performs complete top-down analysis for one symbol:
        
        1. Fetch H4 data → get_market_structure() → get_regime()
        2. Check H4 bias — if NEUTRAL/UNCLEAR, return None early
        3. Fetch H1 data → detect BOS/CHoCH
        4. Check H1 structure aligns with H4 bias
        5. Fetch M15 data → detect liquidity sweeps, OBs, FVGs
        6. Check M15 has valid setup in H4 bias direction
        7. Fetch M5 data → check for M5 confirmation
        8. Assemble TradeSetup if minimum conditions met
        
        Returns TradeSetup if potential setup found, None otherwise.
        Logs analysis results at each step.
        """

    def scan_all_symbols(self) -> list[TradeSetup]:
        """
        Calls analyze_symbol() for all configured pairs.
        Returns list of potential setups (may be empty).
        """

IMPORTANT PERFORMANCE NOTE:
  H4 data is cached — only re-analyzed when H4 bar changes.
  H1 data is cached — only re-analyzed when H1 bar changes.
  M15 and M5 are analyzed every scan cycle.
  This matches the efficient scan cycle requirement.

NO LOOKAHEAD RULE:
  All data slices passed to strategy functions must end at the current bar.
  Never pass future bars to analysis functions.

================================================================================
M5 ENTRY CONFIRMATION — EXACT DEFINITION (DECISION-022)
================================================================================

m5_confirmation = True if ANY ONE of the following conditions is met,
evaluated on the M5 timeframe data for the signal's symbol:

CONDITION A — M5 BOS:
  Most recent CLOSED M5 candle's close price breaks:
    For BUY signal:  close > previous M5 swing HIGH price
    For SELL signal: close < previous M5 swing LOW price
  Uses the SAME close-confirmation rule as Task 05-02 (no intrabar wicks).
  The M5 swing high/low is detected with SWING_LOOKBACK_CANDLES=2 (same config).

CONDITION B — M5 Displacement:
  Most recent CLOSED M5 candle satisfies ALL 4 displacement criteria from
  Task 05-06 (body > 1.5x ATR, body > 70% of range, close in upper/lower 25%,
  DISPLACEMENT_CLOSE_RATIO threshold) in the signal direction:
    For BUY signal:  bullish displacement candle
    For SELL signal: bearish displacement candle

CONDITION C — M5 CHoCH:
  A CHoCH in the signal direction detected on M5 within the last
  M5_CONFIRMATION_LOOKBACK_CANDLES candles:
    For BUY signal:  bearish-to-bullish CHoCH on M5
    For SELL signal: bullish-to-bearish CHoCH on M5
  Uses same close-confirmation rule as Task 05-02.

If NONE of A, B, C are present: m5_confirmation = False

m5_confirmation_type: "BOS" | "DISPLACEMENT" | "CHoCH" | "NONE"
  Priority if multiple conditions met: BOS > DISPLACEMENT > CHoCH

CONFIG (add to app/config.py):
  M5_CONFIRMATION_LOOKBACK_CANDLES = 5  (default — look back 5 M5 bars)

IMPLEMENTATION NOTE:
  Reuse the same detect_structure_breaks() from bos_choch.py on M5 data.
  Reuse the same displacement check from displacement.py on the M5 candle.
  Do NOT reimplement this logic — inject M5 data into existing functions.
